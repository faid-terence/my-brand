{

    const client=await res2.json();
    let temple='';
    client.forEach(post => {
        temple += `
        <div class="card">
        <i class= "${post.icon}"></i>
        <h5>${post.title}</h5>

    })
    blogsContainer.innerHTML=temple;

}
